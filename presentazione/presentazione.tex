% 3. HEADER & FOOTER
% =========================================================
\setbeamertemplate{frametitle}{
    \nointerlineskip
    \begin{beamercolorbox}[wd=\paperwidth, ht=1.3cm, dp=0.4cm, leftskip=0.8cm]{frametitle}
        \raisebox{0.3cm}{\textbf{\Large \insertframetitle}}
        \tikz[overlay, remember picture] {
            \shade[left color=VividViolet, right color=CyanGlow] 
            (0, 0.05) rectangle (\paperwidth, 0.1);
        }
    \end{beamercolorbox}
}

\setbeamertemplate{footline}{
    \leavevmode%
    \hbox{%
        \begin{beamercolorbox}[wd=.33\paperwidth,ht=2.5ex,dp=1.125ex,leftskip=.5cm]{author in head/foot}%
            \color{gray}\scriptsize\insertshortauthor
        \end{beamercolorbox}%
        \begin{beamercolorbox}[wd=.33\paperwidth,ht=2.5ex,dp=1.125ex,center]{title in head/foot}%
            \color{gray!60}\scriptsize\insertshorttitle
        \end{beamercolorbox}%
        \begin{beamercolorbox}[wd=.33\paperwidth,ht=2.5ex,dp=1.125ex,rightskip=.5cm]{date in head/foot}%
            \hfill \textbf{\color{TechViolet}\insertframenumber} \color{gray!40}/ \inserttotalframenumber
        \end{beamercolorbox}%
    }%
    \begin{tikzpicture}[overlay, remember picture]
        \draw[gray!10, fill=gray!10] (current page.south west) rectangle (current page.south east);
        \ifnum\inserttotalframenumber>0
            \fill[VividViolet] (current page.south west) rectangle 
            ($(current page.south west)!\insertframenumber/\inserttotalframenumber!(current page.south east) + (0, 2pt)$);
        \fi
    \end{tikzpicture}
    \vskip0pt%
}

% =========================================================
% 4. TRANSIZIONI SEZIONI
% =========================================================
\AtBeginSection[]{
    {
        \setbeamercolor{background canvas}{bg=DeepBlack} 
        \begin{frame}[plain, noframenumbering]
            \centering
            \begin{tikzpicture}
                \node[circle, fill=VividViolet, opacity=0.10, minimum size=5.5cm] at (0,0) {};
                \node[circle, fill=VividViolet, opacity=0.20, minimum size=4.8cm] at (0,0) {};
                \node[circle, draw=VividViolet, line width=2.5pt, minimum size=4cm] (ring) at (0,0) {};
                
                \node[text=CyanGlow, font=\fontsize{80}{90}\selectfont\bfseries] at (0,0) {\thesection};
                
                \node[below=2.0cm of ring, text=PaperWhite, font=\fontsize{24}{28}\selectfont\bfseries, align=center] (title) {\insertsection};
                
                \draw[NeonPink, thick] ($(title.south west) + (-1, -0.2)$) -- ($(title.south east) + (1, -0.2)$);
            \end{tikzpicture}
        \end{frame}
    }
}
